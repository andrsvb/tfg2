\begin{question}{test_suma}
    ¿Cuál es el resultado de la suma: $\eval{a} + \eval{b}$?
    \begin{choices}
        \wrongchoice{\eval{a+b+1}}
        \correctchoice{\eval{a+b}}
        \wrongchoice{\eval{a+b-1}}
        \wrongchoice{\eval{a+b-2}}
    \end{choices}
\end{question}


\begin{question}{test_par}
    ¿El resultado de la suma es par o impar?
    \begin{choices}
        \correctchoice{\sif{divmod((a + b), 2)[1] == 0}{Par}{Impar}}
        \wrongchoice{\sif{divmod((a + b), 2)[1] == 0}{Impar}{Par}}
    \end{choices}
\end{question}
